scp -r fhv xl1638@dumbo.es.its.nyu.edu:./
3, 
//log into nyu dumbo
ssh -Y xl1638@dumbo.es.its.nyu.edu
4, 
//make a directory in hdfs to hold files
hdfs dfs -mkdir /user/xl1638/spark_final_project
5, 
//copy data from dumbo to hdfs
hdfs dfs -put fhv /user/xl1638/spark_final_project\
6, 
//check whether the data are there
hdfs dfs -ls /user/xl1638/spark_final_project/fhv
